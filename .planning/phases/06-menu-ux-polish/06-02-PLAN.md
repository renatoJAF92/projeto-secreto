### T-06-02-01: Gerar assets de menu via Python
**File:** `assets/sprites/ui/menu_background.png` e `assets/audio/music/menu_theme.ogg`

Script Python gerador (`tools/gen_menu_assets.py`):

```python
#!/usr/bin/env python3
"""Generates menu_background.png (320x180 pixel art cityscape) and menu_theme.ogg (simple melody)."""
import struct, math, os

# --- menu_background.png: cidade noturna simples ---
def write_png(path, width, height, pixels):
    import zlib
    def u32(n): return struct.pack('>I', n)
    def chunk(tag, data):
        c = zlib.crc32(tag + data) & 0xFFFFFFFF
        return u32(len(data)) + tag + data + u32(c)
    raw = b''
    for row in pixels:
        raw += b'\x00' + bytes(row)
    compressed = zlib.compress(raw, 9)
    ihdr = struct.pack('>IIBBBBB', width, height, 8, 2, 0, 0, 0)
    with open(path, 'wb') as f:
        f.write(b'\x89PNG\r\n\x1a\n')
        f.write(chunk(b'IHDR', ihdr))
        f.write(chunk(b'IDAT', compressed))
        f.write(chunk(b'IEND', b''))

W, H = 320, 180
SKY_TOP    = (10, 10, 35)
SKY_BOT    = (20, 20, 60)
BLDG_FAR   = (18, 18, 45)
BLDG_MID   = (25, 25, 65)
BLDG_NEAR  = (15, 15, 40)
WINDOW_ON  = (255, 220, 100)
WINDOW_OFF = (20, 20, 50)
GROUND     = (12, 12, 30)
STAR       = (200, 200, 255)

import random
rng = random.Random(42)

pixels = []
for y in range(H):
    row = []
    t = y / H
    r = int(SKY_TOP[0] * (1-t) + SKY_BOT[0] * t)
    g = int(SKY_TOP[1] * (1-t) + SKY_BOT[1] * t)
    b = int(SKY_TOP[2] * (1-t) + SKY_BOT[2] * t)
    for x in range(W):
        row.extend([r, g, b])
    pixels.append(row)

# Stars
for _ in range(80):
    sx, sy = rng.randint(0, W-1), rng.randint(0, H//2)
    pixels[sy][sx*3:sx*3+3] = list(STAR)

def draw_building(bx, bw, bh, color, win_color_on, win_color_off):
    by = H - bh
    for y in range(by, H):
        for x in range(bx, bx+bw):
            if 0 <= x < W:
                pixels[y][x*3:x*3+3] = list(color)
    for y in range(by+4, H-4, 8):
        for x in range(bx+3, bx+bw-3, 6):
            if 0 <= x < W:
                wc = win_color_on if rng.random() > 0.4 else win_color_off
                pixels[y][x*3:x*3+3] = list(wc)

# Far buildings
for i in range(8):
    bx = i * 40 + rng.randint(-5, 5)
    bw = rng.randint(28, 44)
    bh = rng.randint(60, 100)
    draw_building(bx, bw, bh, BLDG_FAR, WINDOW_ON, WINDOW_OFF)

# Mid buildings
for i in range(6):
    bx = i * 52 + rng.randint(0, 10)
    bw = rng.randint(36, 52)
    bh = rng.randint(40, 80)
    draw_building(bx, bw, bh, BLDG_MID, WINDOW_ON, WINDOW_OFF)

# Near buildings (dark silhouette)
for i in range(4):
    bx = i * 80 + rng.randint(-10, 10)
    bw = rng.randint(60, 72)
    bh = rng.randint(25, 45)
    draw_building(bx, bw, bh, BLDG_NEAR, WINDOW_OFF, WINDOW_OFF)

# Ground strip
for y in range(H-8, H):
    for x in range(W):
        pixels[y][x*3:x*3+3] = list(GROUND)

os.makedirs('assets/sprites/ui', exist_ok=True)
write_png('assets/sprites/ui/menu_background.png', W, H, pixels)
print("Generated assets/sprites/ui/menu_background.png")

# --- menu_theme.ogg: melodia simples ---
# Generate WAV first, then convert to OGG via Python wave + audioop
import wave, io

SAMPLE_RATE = 44100
DURATION = 32  # seconds (loops cleanly)
BPM = 110
BEAT = 60.0 / BPM

# Simple pentatonic melody (MIDI-like note list: (beat_offset, duration_beats, freq))
NOTES = [
    (0.0, 0.5, 523.25),   # C5
    (0.5, 0.5, 587.33),   # D5
    (1.0, 0.5, 659.25),   # E5
    (1.5, 0.5, 783.99),   # G5
    (2.0, 1.0, 880.00),   # A5
    (3.0, 0.5, 783.99),   # G5
    (3.5, 0.5, 659.25),   # E5
    (4.0, 0.5, 587.33),   # D5
    (4.5, 0.5, 523.25),   # C5
    (5.0, 1.0, 440.00),   # A4
    (6.0, 0.5, 523.25),   # C5
    (6.5, 0.5, 659.25),   # E5
    (7.0, 1.0, 783.99),   # G5
    (8.0, 0.5, 880.00),   # A5
    (8.5, 0.5, 783.99),   # G5
    (9.0, 0.5, 659.25),   # E5
    (9.5, 0.5, 523.25),   # C5
    (10.0, 2.0, 440.00),  # A4
]

samples = [0.0] * (SAMPLE_RATE * DURATION)

def add_note(samples, start_sec, dur_sec, freq, amplitude=0.25):
    start = int(start_sec * SAMPLE_RATE)
    end = int((start_sec + dur_sec) * SAMPLE_RATE)
    attack = int(0.01 * SAMPLE_RATE)
    release = int(0.05 * SAMPLE_RATE)
    for i in range(end - start):
        t = i / SAMPLE_RATE
        env = 1.0
        if i < attack:
            env = i / attack
        elif i > (end - start) - release:
            env = ((end - start) - i) / release
        s = start + i
        if s < len(samples):
            samples[s] += amplitude * env * math.sin(2 * math.pi * freq * t)
            samples[s] += amplitude * 0.3 * env * math.sin(4 * math.pi * freq * t)  # harmonic

# Repeat melody 3x to fill 32s
for rep in range(3):
    offset = rep * 12 * BEAT
    for (beat, dur, freq) in NOTES:
        add_note(samples, offset + beat * BEAT, dur * BEAT, freq)

# Normalize and convert to 16-bit PCM
peak = max(abs(s) for s in samples) or 1.0
pcm = bytes()
for s in samples:
    v = int(s / peak * 28000)
    v = max(-32768, min(32767, v))
    pcm += struct.pack('<h', v)

wav_buf = io.BytesIO()
with wave.open(wav_buf, 'wb') as wf:
    wf.setnchannels(1)
    wf.setsampwidth(2)
    wf.setframerate(SAMPLE_RATE)
    wf.writeframes(pcm)

wav_bytes = wav_buf.getvalue()

# Write WAV first (Godot can use WAV directly; OGG conversion requires external tool)
os.makedirs('assets/audio/music', exist_ok=True)
wav_path = 'assets/audio/music/menu_theme.wav'
with open(wav_path, 'wb') as f:
    f.write(wav_bytes)
print(f"Generated {wav_path}")
print("NOTE: Convert to OGG with: ffmpeg -i menu_theme.wav -c:a libvorbis menu_theme.ogg")
print("      OR use the WAV directly (rename reference in main_menu.gd to .wav)")
```

### T-06-02-03: Atualizar main_menu.gd — lógica de popup customizado + música
**File:** `scenes/main_menu/main_menu.gd`

```gdscript
extends Control

@onready var continue_button: Button = $ButtonGroup/ContinueButton
@onready var new_game_button: Button = $ButtonGroup/NewGameButton
@onready var options_button: Button = $ButtonGroup/OptionsButton
@onready var confirm_panel: Panel = $ConfirmPanel
@onready var confirm_btn: Button = $ConfirmPanel/ButtonRow/ConfirmButton
@onready var cancel_btn: Button = $ConfirmPanel/ButtonRow/CancelButton
@onready var background_image: TextureRect = $BackgroundImage

const _MENU_MUSIC := preload("res://assets/audio/music/menu_theme.ogg")
# Se .ogg não existir, usar: preload("res://assets/audio/music/menu_theme.wav")


func _ready() -> void:
    # Background
    if ResourceLoader.exists("res://assets/sprites/ui/menu_background.png"):
        background_image.texture = load("res://assets/sprites/ui/menu_background.png")

    # Music
    AudioManager.play_music(_MENU_MUSIC)

    # Buttons
    continue_button.disabled = not SaveManager.save_exists()
    if continue_button.disabled:
        continue_button.add_theme_color_override("font_color", Color("#888888"))
        continue_button.add_theme_color_override("font_disabled_color", Color("#888888"))

    continue_button.pressed.connect(_on_continue_pressed)
    new_game_button.pressed.connect(_on_new_game_pressed)
    options_button.pressed.connect(_on_options_pressed)
    confirm_btn.pressed.connect(_on_new_game_confirmed)
    cancel_btn.pressed.connect(func(): confirm_panel.visible = false)

    if SaveManager.save_exists():
        continue_button.grab_focus()
    else:
        new_game_button.grab_focus()


func _on_continue_pressed() -> void:
    AudioManager.stop_music()
    SceneTransition.go_to(SaveManager.get_checkpoint_scene())


func _on_new_game_pressed() -> void:
    if SaveManager.save_exists():
        confirm_panel.visible = true
        confirm_btn.grab_focus()
        return
    SaveManager.new_game()
    AudioManager.stop_music()
    SceneTransition.go_to("res://scenes/world1/mundo1_abertura.tscn")


func _on_new_game_confirmed() -> void:
    confirm_panel.visible = false
    SaveManager.new_game()
    AudioManager.stop_music()
    SceneTransition.go_to("res://scenes/world1/mundo1_abertura.tscn")


func _on_options_pressed() -> void:
    if ResourceLoader.exists("res://scenes/options_menu/options_menu.tscn"):
        SceneTransition.go_to("res://scenes/options_menu/options_menu.tscn")
```

**Nota sobre preload:** O `preload()` falha se o arquivo não existir em tempo de parse. Usar uma constante com `preload` é seguro apenas se o arquivo for garantido. Para robustez, a texture do background usa `ResourceLoader.exists()` guard antes de carregar. Para a música, se o arquivo OGG não existir, substituir por `.wav` na constante.
